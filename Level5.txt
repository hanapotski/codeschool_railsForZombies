Route Parameters
/app/controllers/tweets_controller.rb

def index
  if params[:zipcode]
    @tweets = Tweet.where(zipcode: params[:zipcode])
  else
   @tweets = Tweet.all
  end
  respond_to do |format|
   format.html # index.html.erb
   format.xml { render xml: @tweets }
   end
 end
